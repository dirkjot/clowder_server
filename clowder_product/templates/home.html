You have a script you need to run monthly.

Just add ~1 line of code..

Python - Node - Ruby - PHP 
import clowder
import datetime

clowder.ok(
    name='Monthly Job',
    schedule=datetime.timedelta(days=31),
)



require 'clowder.php'

$clowder->ok([
    'name' => 'Monthly Job',
    'schedule' => new DateInterval('P31D')
]);



require "clowder"
require "duration"

clowder.ok(
    name: 'Monthly Job',
    schedule: Duration.new(:days => 5)
)



var clowder = require('clowder');
var later = require('later');

clowder.ok({
    'name': 'Monthly Job',
    'schedule': later.recur().every().month(),
});

You want an alert when CPU usage goes over 60%.

Just add ~1 line of code.

Python - Node - Ruby - PHP 
import clowder
import psutil

cpu_percentage = psutil.cpu_percent()

clowder.submit(
    name='CPU Percentage',
    alert=lambda x: x >= 60,
    value=cpu_percentage
)



require 'clowder.php'

$load = sys_getloadavg();

$clowder->ok([
    'name' => 'CPU Percentage',
    'value' => $load[0],
    'alert' => function (x) {return x >= 60;}
]);



require "clowder"
require "duration"

clowder.ok(
    name: 'CPU Percentage',
    schedule: Duration.new(:days => 5)
)



var clowder = require('clowder');
var later = require('later');

clowder.ok({
    'name': 'CPU Percentage',
    'schedule': later.recur().every().month(),
});
